// Core logic: RFID Attendance Marking
#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 10
#define RST_PIN 9
MFRC522 rfid(SS_PIN, RST_PIN);

String knownID = "AB CD EF 12"; // Example tag UID
void setup() {
  Serial.begin(9600);
  SPI.begin();
  rfid.PCD_Init();
  Serial.println("Scan RFID Tag...");
}

void loop() {
  if (!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial()) return;
  String tagID = "";
  for (byte i = 0; i < rfid.uid.size; i++) {
    tagID += String(rfid.uid.uidByte[i], HEX);
  }
  tagID.toUpperCase();
  Serial.print("Tag detected: "); Serial.println(tagID);

  if (tagID == knownID) {
    Serial.println("Attendance marked!");
  } else {
    Serial.println("Unauthorized tag.");
  }
  delay(1000);
}
